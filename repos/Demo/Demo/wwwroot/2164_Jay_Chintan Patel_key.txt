using Microsoft.AspNetCore.Hosting;
using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Mvc;
using System.IO;
using System.Threading.Tasks;

public class DoctorController : Controller
{
    private readonly IWebHostEnvironment _hostEnvironment;
    private readonly YourDbContext _db; // Replace with your DbContext type

    public DoctorController(IWebHostEnvironment hostEnvironment, YourDbContext db)
    {
        _hostEnvironment = hostEnvironment;
        _db = db;
    }

    [HttpPost]
    public async Task<IActionResult> UploadDocuments(IEnumerable<IFormFile> files, int userId)
    {
        foreach (var file in files)
        {
            if (file != null && file.Length > 0)
            {
                var fileName = Path.GetFileName(file.FileName);
                var uploads = Path.Combine(_hostEnvironment.WebRootPath, "documents", fileName);

                using (var fileStream = new FileStream(uploads, FileMode.Create))
                {
                    await file.CopyToAsync(fileStream);
                }

                Documents docModel = new Documents()
                {
                    UserId = userId,
                    Document_name = fileName,
                    Document_path = @"\documents\",
                };

                switch (Path.GetExtension(fileName))
                {
                    case ".doc":
                    case ".docx":
                        docModel.Document_type = "docx";
                        break;
                    case ".xls":
                    case ".xlsx":
                        docModel.Document_type = "xlsx";
                        break;
                    case ".pdf":
                        docModel.Document_type = "pdf";
                        break;
                    default:
                        break;
                }

                _db.Documents.Add(docModel);
                await _db.SaveChangesAsync();
            }
        }

        return Content("Files uploaded successfully!");
    }
}
